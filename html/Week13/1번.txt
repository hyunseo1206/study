    <script type = "text/javascript">
        function drawCircle(canvas, x, y, radius, color) {
            canvas.beginPath();                             // 경로 지정을 시작하는 메소드
            canvas.arc(x, y, radius, 0, 2 * Math.PI);       // 원호 그리기
            canvas.lineWidth = 10;                          // 선 두께는 10 픽셀로 설정
            canvas.strokeStyle = color;                     // 선의 색상을 지정
            canvas.stroke();                                // 현재 지정된 색상과 선 끝 모양으로 선을 그림.
        }

        function drawCircle2(canvas, x, y, radius, color) {
            canvas.beginPath();                             
            canvas.arc(x, y, radius, -1, 0.1 * Math.PI);       
            canvas.lineWidth = 10;                          
            canvas.strokeStyle = color;                    
            canvas.stroke(); 
        }

        function drawCircle3(canvas, x, y, radius, color) {
            canvas.beginPath();                             
            canvas.arc(x, y, radius, 0.5 * Math.PI, 0.7 * Math.PI);       
            canvas.lineWidth = 10;                          
            canvas.strokeStyle = color;                    
            canvas.stroke(); 
        }

        function drawOlympicsRings(canvas, x, y) {
            // 시작 지점과 반지름 설정하기
            var startX = 80;
            var startY = 80;
            var radius = 50;

            //색 지정과 변수 전환
            var colors=["blue", "black", "red", "yellow", "green"];
            var x = startX;
            var y = startY;
            var gap = 25;

            //왼쪽 위의 원을 그리기를 위한 for문
            // (60, 60) 에서 반지름이 25인 원을 그리기 시작함.
            for(i = 0; i < 3; i++) {
                drawCircle(canvas, x, y, radius, colors[i]);
                x += radius * 2 + gap;
            }

            //아래 원을 그리기 위한 좌표 재설정
            x = startX + radius + gap/2;
            y = startY + radius/2 + gap;

            //왼쪽 아래의 원을 그리기를 위한 for문    
            for(j = i; j < 5; j++) {
                drawCircle(canvas, x, y, radius, colors[j]);
                x += radius * 2 + gap;
            }

            // 겹치는 부분을 지정하여서 다시 그려주기
            var x = startX;
            var y = startY;
            
            for (i = 0; i < 3; i++) {
                drawCircle2(canvas, x, y, radius, colors[i]);
                drawCircle3(canvas, x, y, radius, colors[i]);
                x += radius * 2 + gap;
            }
        }
    </script>

    <body>
        <!--캔버스 설정 및 그리기-->
        <canvas id = "myCanvas" width = "800" height = "400"></canvas>
        <script type = "text/javascript">
            var canvas = document.getElementById("myCanvas");
            var context = canvas.getContext("2d");
            drawOlympicsRings(context);

            
            context.font="bold 25pt Arial";
            context.fillStyle = "black";
            context.fillText("올림픽 오륜기", 100, 250);

        </script>
